\section{Model}
\begin{frame}
    \begin{columns}
        \begin{column}{0.6\textwidth}
            \begin{equation*}
                \pi_{\tech{i}}(t) = a_{\tech{i}} N(t)^{b_{\tech{i}}}
            \end{equation*} \uncover<2->{where $\tech{i} = 1$ if the firm has adopted an adaptation technology at time $t$.}
        \end{column}
        \begin{column}{0.4\textwidth}
            \begin{itemize}
                \item Profits $\pi_{\tech{i}}(t)$,
                \item natural resource $N(t)$,
                \item profitability $a_{\tech{i}} > 0$,
                \item dependence $b_{\tech{i}}$ < 1.
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}
    Technology makes the firm \begin{align*}
        \text{more profitable } a_{\tech{1}} &> a_{\tech{0}} \\
        \text{less susceptible } b_{\tech{1}} &< b_{\tech{0}}.
    \end{align*}
    
    \uncover<2>{\begin{figure}
        \input{figures/profit.tikz}
    \end{figure}
    }
\end{frame}

\begin{frame}
    The firm decides the time $\con{\tau}$ at which it should adopt this technology at a cost $I$.
\end{frame}

\begin{frame}
    Firms believe the future natural resource $N(t)$ follows \begin{equation*}
        \frac{\dd{N}(t)}{N(t)} = \alpha \dd{t} + \sigma \dd{B(t)},
    \end{equation*} but they do not know $\alpha$ and $\sigma^2$. 
    
    \uncover<2>{
        A belief is simply \begin{equation*}
            (\alpha, \sigma^2) = \theta \in \Theta \coloneqq [\underline{\alpha}, \overline{\alpha}] \times [\underline{\sigma}, \overline{\sigma}].
        \end{equation*}
    } 
\end{frame}


\begin{frame}
    Some technicalities: \begin{enumerate}\setlength{\itemsep}{1em}
        \item Each belief $\theta$ induces a probability measure $P^\theta$ on the domain of $N$.
        \pause \item Call $P^{\Theta} \coloneqq \{P^{\theta} : \theta \in \Theta\}$.
        \pause \item Index beliefs by ${\bel{q}} \in P^{\Theta}$ such that \begin{equation*}
            \frac{\dd{N}(t)}{N(t)} = \alpha_{\bel{q}} \dd{t} + \sigma_{\bel{q}} \dd{B(t)},
        \end{equation*}
    \end{enumerate}
\end{frame}


\begin{frame}
    Combining \textbf{beliefs} and \textbf{technology}, we get the expected profit dynamics \begin{equation*}
        \frac{\dd{\pi_{\tech{i}, \bel{q}}(t)}}{\pi_{\tech{i}, \bel{q}}(t)} = \underbrace{\left( b_{\tech{i}} \alpha_{\bel{q}} + \frac{1}{2} b_{\tech{i}} (b_{\tech{i}} - 1) \sigma_{\bel{q}}^2 \right)}_{\mu_{\tech{i}, \bel{q}}} \dd{t} + \underbrace{b_{\tech{i}} \sigma_{\bel{q}}}_{\Sigma_{\tech{i}, \bel{q}}} \dd{B}(t).
    \end{equation*}
\end{frame}

\begin{frame}
    \begin{figure}
        \resizebox{!}{0.95\paperheight}{\input{figures/profitevolution.tikz}}
    \end{figure}
\end{frame}

\begin{frame}
    The \textbf{belief} of the firm comes into the expectations used for the objective \begin{equation*}
        \mathbb{B}_{\varepsilon, \bel{q}}[\pi] \coloneqq \mathbb{E}\left[\pi \Big \vert \underbrace{\varepsilon \bel{q} + (1 - \varepsilon) \delta}_{\text{behavioural weight}}  \right]
    \end{equation*} where $\delta$ is the empirical data and $\varepsilon$ is a behavioural distortion or ambiguity aversion.
\end{frame}

\begin{frame}
    The firm problem is then \begin{equation*}
        \begin{split}
            V(\pi_{\tech{i}, q}(t)) = \sup_{\con{\tau}} \inf_{\bel{q} \in P^{\Theta}} \Bigg\{ &\mathbb{B}_{\varepsilon, \bel{q}} \Bigg[ \overbrace{\int^{\con{\tau}}_{t} \pi_{\tech{0}, \bel{q}}(s) e^{-\rho (s - t)} \dd{s}}^{\text{pre-investment}} \\
            \uncover<2->{&\underbrace{\left( \int^{\infty}_{\con{\tau}} \pi_{\tech{1}, \bel{q}}(s) e^{-\rho (s - \con{\tau})} \dd{s} - I \right) e^{-\rho(\con{\tau} - t)}}_{\text{post-investment}} \Bigg]}
        \end{split}
    \end{equation*}
\end{frame}



\begin{frame}
    The firm switches whenever $N(t)$ falls some $N^*$. No analytical expression but nice properties: \begin{enumerate}
        \item Effect of risk $\frac{\partial N^*}{\partial \sigma}$ ambiguous \begin{itemize}
            \item Increased opportunity costs $N^* \downarrow$ \\
            \item Increases discounting $N^* \uparrow$
        \end{itemize}
        \item Effect of ambiguity $\frac{\partial N^*}{\partial \varepsilon}$ ambiguous \begin{itemize}
            \item Increased opportunity costs $N^* \downarrow$ \\
            \item Reduced benefits of technology through discounting $N^* \uparrow$
        \end{itemize}
    \end{enumerate}
\end{frame}

\section{What can we learn for BELIEFS?}

\begin{frame}
    It seems to me like ``our'' problem is analogous \begin{table}
        \centering
        \footnotesize
        \begin{tabular}{l|| c | c}
            & Yan \& Sims & Beliefs \\ \hline\hline
            $N(t)$ & Natural resource & Climate policy / carbon price \\ \hline
            $\pi(t)$ & Profits of firms & Profits of firms \\ \hline
            $\tech{i}$ & Reduces resource dependence & Reduces exposure to carbon price \\ \hline
            $\bel{q}$ & Belief about future natural resource & Belief about future policy
        \end{tabular}
    \end{table}
\end{frame}
\begin{frame}
    If so, some lessons \begin{enumerate} \setlength{\itemsep}{0.5em}
        \item Should we think of ``beliefs'' in the same way? No need to estimate future beliefs $\bel{q}$, but only ambiguity aversion $\varepsilon$ and possible futures $\Theta$.
        \pause \item Uncertainty around drift $\alpha_{\bel{q}}$ and noise $\sigma_{\bel{q}}$ have opposite effects!
        \pause \item We cannot think of beliefs $\bel{q}$ independently of technology $a_{\tech{i}}, b_{\tech{i}}$ (old tale \textcite{gilboaTheoryDecisionUncertainty2009}) 
        \pause \item Irreversibility $I$ matters (old tale \textcite{stokeyEconomicsInactionStochastic2008})
    \end{enumerate}
\end{frame}

\begin{frame}
    One key difference: \begin{itemize}
        \item[] \textcite{yan2025}'s firms are facing nature (at least in the model). 
        \item[] ``Our'' firms face a policymaker.
    \end{itemize}
\end{frame}
